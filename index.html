                            public static function ConvertToDB($arProperty, $value)
                            {
                                $arSettings = static::PrepareSettings($arProperty);
                                if (isset($arSettings['USER_TYPE_SETTINGS']))
                                    $arSettings = $arSettings['USER_TYPE_SETTINGS'];

                                if(isset($value["VALUE"]["INPUT_VALUE"]) == TRUE){
                                    if (CModule::IncludeModule('highloadblock') && ($value["VALUE"]["INPUT_VALUE"][9] || $value["VALUE"]["INPUT_VALUE"][2]["error"] == 0 || $value["VALUE"]["INPUT_VALUE"][4] || $value["VALUE"]["INPUT_VALUE"][5] || $value["VALUE"]["INPUT_VALUE"][6] || $value["VALUE"]["INPUT_VALUE"][7])){
                                        if($arSettings['HIGHLOAD_ID']){
                                            $HIGHLOAD_ID = $arSettings['HIGHLOAD_ID'];
                                        }else{
                                            $HIGHLOAD_ID = 2;
                                        }
                                        $hlblock = HL\HighloadBlockTable::getById($HIGHLOAD_ID)->fetch(); 

                                        $entity = HL\HighloadBlockTable::compileEntity($hlblock); 
                                        $entity_data_class = $entity->getDataClass(); 

                                        if($value["VALUE"]["INPUT_VALUE"][1] == "on"){
                                            $data = array(
                                                "UF_CENTER" => 1,
                                                "UF_TITLE_LEFT" => $value["VALUE"]["INPUT_VALUE"][4],
                                                "UF_TEXT_LEFT" => $value["VALUE"]["INPUT_VALUE"][5],
                                                "UF_TITLE_RIGHT" => "",
                                                "UF_TEXT_RIGHT" => "",
                                            );
                                        }else{
                                            $data = array(
                                                "UF_CENTER" => 0,
                                                "UF_TITLE_LEFT" => $value["VALUE"]["INPUT_VALUE"][4],
                                                "UF_TEXT_LEFT" => $value["VALUE"]["INPUT_VALUE"][5],
                                                "UF_TITLE_RIGHT" => $value["VALUE"]["INPUT_VALUE"][6],
                                                "UF_TEXT_RIGHT" => $value["VALUE"]["INPUT_VALUE"][7],
                                            );
                                        }
                                        
                                        if(isset($value["VALUE"]["INPUT_VALUE"][11]) == FALSE && $value["VALUE"]["INPUT_VALUE"][1] != "on"){ // right
                                            if($value["VALUE"]["INPUT_VALUE"][8] && $value["VALUE"]["INPUT_VALUE"][3]["error"] > 0){// если есть старая загруженная картинка и нет новой
                                            }else{
                                                $data["UF_PIC_RIGHT"] = $value["VALUE"]["INPUT_VALUE"][3];
                                            }
                                        }else{
                                            $data["UF_PIC_RIGHT"] = "";
                                        }

                                        if(isset($value["VALUE"]["INPUT_VALUE"][10]) == FALSE){ // left
                                            if($value["VALUE"]["INPUT_VALUE"][9] && $value["VALUE"]["INPUT_VALUE"][2]["error"] > 0){            
                                            }else{
                                                $data["UF_PIC_LEFT"] = $value["VALUE"]["INPUT_VALUE"][2];
                                            }
                                        }else{
                                            $data["UF_PIC_LEFT"] = "";
                                        }
                                        if($value["VALUE"]["INPUT_VALUE"][0] > 0){// if there is id from HighloadBlock
                                            $result_update = $entity_data_class::update($value["VALUE"]["INPUT_VALUE"][0], $data);
                                        }else{
                                            $result = $entity_data_class::add($data);
                                            $value = $result->getId();
                                        }
                                    }
                                    if(!$result_update && !empty($value) && empty($value["VALUE"]["INPUT_VALUE"])){
                                        return $value;
                                    }elseif($result_update){
                                        return $value["VALUE"]["INPUT_VALUE"][0];
                                    }
                                }else{
                                    return $value;
                                }
                            }
